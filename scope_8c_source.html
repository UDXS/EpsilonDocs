43 scope_t *scope_new(scope_kind_t kind, mp_parse_node_t pn, qstr source_file, mp_uint_t emit_options) {
44  scope_t *scope = m_new0(scope_t, 1);
45  scope->kind = kind;
46  scope->pn = pn;
47  scope->source_file = source_file;
48  if (kind == SCOPE_FUNCTION || kind == SCOPE_CLASS) {
49  assert(MP_PARSE_NODE_IS_STRUCT(pn));
50  scope->simple_name = MP_PARSE_NODE_LEAF_ARG(((mp_parse_node_struct_t*)pn)->nodes[0]);
51  } else {
52  scope->simple_name = scope_simple_name_table[kind];
53  }
54  scope->raw_code = mp_emit_glue_new_raw_code();
55  scope->emit_options = emit_options;
56  scope->id_info_alloc = MICROPY_ALLOC_SCOPE_ID_INIT;
57  scope->id_info = m_new(id_info_t, scope->id_info_alloc);
58 
59  return scope;
60 }

67 id_info_t *scope_find_or_add_id(scope_t *scope, qstr qst, bool *added) {
68  id_info_t *id_info = scope_find(scope, qst);
69  if (id_info != NULL) {
70  *added = false;
71  return id_info;
72  }
73 
74  // make sure we have enough memory
75  if (scope->id_info_len >= scope->id_info_alloc) {
76  scope->id_info = m_renew(id_info_t, scope->id_info, scope->id_info_alloc, scope->id_info_alloc + MICROPY_ALLOC_SCOPE_ID_INC);
77  scope->id_info_alloc += MICROPY_ALLOC_SCOPE_ID_INC;
78  }
79 
80  // add new id to end of array of all ids; this seems to match CPython
81  // important thing is that function arguments are first, but that is
82  // handled by the compiler because it adds arguments before compiling the body
83  id_info = &scope->id_info[scope->id_info_len++];
84 
85  id_info->kind = 0;
86  id_info->flags = 0;
87  id_info->local_num = 0;
88  id_info->qst = qst;
89  *added = true;
90  return id_info;
91 }

93 id_info_t *scope_find(scope_t *scope, qstr qst) {
94  for (mp_uint_t i = 0; i < scope->id_info_len; i++) {
95  if (scope->id_info[i].qst == qst) {
96  return &scope->id_info[i];
97  }
98  }
99  return NULL;
100 }

109 STATIC void scope_close_over_in_parents(scope_t *scope, qstr qst) {
110  assert(scope->parent != NULL); // we should have at least 1 parent
111  for (scope_t *s = scope->parent;; s = s->parent) {
112  assert(s->parent != NULL); // we should not get to the outer scope
113  bool added;
114  id_info_t *id = scope_find_or_add_id(s, qst, &added);
115  if (added) {
116  // variable not previously declared in this scope, so declare it as free and keep searching parents
117  id->kind = ID_INFO_KIND_FREE;
118  } else {
119  // variable is declared in this scope, so finish
120  if (id->kind == ID_INFO_KIND_LOCAL) {
121  // variable local to this scope, close it over
122  id->kind = ID_INFO_KIND_CELL;
123  } else {
124  // ID_INFO_KIND_FREE: variable already closed over in a parent scope
125  // ID_INFO_KIND_CELL: variable already closed over in this scope
126  assert(id->kind == ID_INFO_KIND_FREE || id->kind == ID_INFO_KIND_CELL);
127  }
128  return;
129  }
130  }
131 }

133 void scope_find_local_and_close_over(scope_t *scope, id_info_t *id, qstr qst) {
134  if (scope->parent != NULL) {
135  for (scope_t *s = scope->parent; s->parent != NULL; s = s->parent) {
136  id_info_t *id2 = scope_find(s, qst);
137  if (id2 != NULL) {
138  if (id2->kind == ID_INFO_KIND_LOCAL || id2->kind == ID_INFO_KIND_CELL || id2->kind == ID_INFO_KIND_FREE) {
139  id->kind = ID_INFO_KIND_FREE;
140  scope_close_over_in_parents(scope, qst);
141  return;
142  }
143  break;
144  }
145  }
146  }
147  id->kind = ID_INFO_KIND_GLOBAL_IMPLICIT;
148 }
