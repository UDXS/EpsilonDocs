34 STATIC void module_print(const mp_print_t *print, mp_obj_t self_in, mp_print_kind_t kind) {
35  (void)kind;
36  mp_obj_module_t *self = MP_OBJ_TO_PTR(self_in);
37 
38  const char *module_name = "";
39  mp_map_elem_t *elem = mp_map_lookup(&self->globals->map, MP_OBJ_NEW_QSTR(MP_QSTR___name__), MP_MAP_LOOKUP);
40  if (elem != NULL) {
41  module_name = mp_obj_str_get_str(elem->value);
42  }
43 
44 #if MICROPY_PY___FILE__
45  // If we store __file__ to imported modules then try to lookup this
46  // symbol to give more information about the module.
47  elem = mp_map_lookup(&self->globals->map, MP_OBJ_NEW_QSTR(MP_QSTR___file__), MP_MAP_LOOKUP);
48  if (elem != NULL) {
49  mp_printf(print, "<module '%s' from '%s'>", module_name, mp_obj_str_get_str(elem->value));
50  return;
51  }
52 #endif
53 
54  mp_printf(print, "<module '%s'>", module_name);
55 }

57 STATIC void module_attr(mp_obj_t self_in, qstr attr, mp_obj_t *dest) {
58  mp_obj_module_t *self = MP_OBJ_TO_PTR(self_in);
59  if (dest[0] == MP_OBJ_NULL) {
60  // load attribute
61  mp_map_elem_t *elem = mp_map_lookup(&self->globals->map, MP_OBJ_NEW_QSTR(attr), MP_MAP_LOOKUP);
62  if (elem != NULL) {
63  dest[0] = elem->value;
64  }
65  } else {
66  // delete/store attribute
67  mp_obj_dict_t *dict = self->globals;
68  if (dict->map.is_fixed) {
69  #if MICROPY_CAN_OVERRIDE_BUILTINS
70  if (dict == &mp_module_builtins_globals) {
71  if (MP_STATE_VM(mp_module_builtins_override_dict) == NULL) {
72  MP_STATE_VM(mp_module_builtins_override_dict) = MP_OBJ_TO_PTR(mp_obj_new_dict(1));
73  }
74  dict = MP_STATE_VM(mp_module_builtins_override_dict);
75  } else
76  #endif
77  {
78  // can't delete or store to fixed map
79  return;
80  }
81  }
82  if (dest[1] == MP_OBJ_NULL) {
83  // delete attribute
84  mp_obj_dict_delete(MP_OBJ_FROM_PTR(dict), MP_OBJ_NEW_QSTR(attr));
85  } else {
86  // store attribute
87  // TODO CPython allows STORE_ATTR to a module, but is this the correct implementation?
88  mp_obj_dict_store(MP_OBJ_FROM_PTR(dict), MP_OBJ_NEW_QSTR(attr), dest[1]);
89  }
90  dest[0] = MP_OBJ_NULL; // indicate success
91  }
92 }

101 mp_obj_t mp_obj_new_module(qstr module_name) {
102  mp_map_t *mp_loaded_modules_map = &MP_STATE_VM(mp_loaded_modules_dict).map;
103  mp_map_elem_t *el = mp_map_lookup(mp_loaded_modules_map, MP_OBJ_NEW_QSTR(module_name), MP_MAP_LOOKUP_ADD_IF_NOT_FOUND);
104  // We could error out if module already exists, but let C extensions
105  // add new members to existing modules.
106  if (el->value != MP_OBJ_NULL) {
107  return el->value;
108  }
109 
110  // create new module object
111  mp_obj_module_t *o = m_new_obj(mp_obj_module_t);
112  o->base.type = &mp_type_module;
113  o->globals = MP_OBJ_TO_PTR(mp_obj_new_dict(MICROPY_MODULE_DICT_SIZE));
114 
115  // store __name__ entry in the module
116  mp_obj_dict_store(MP_OBJ_FROM_PTR(o->globals), MP_OBJ_NEW_QSTR(MP_QSTR___name__), MP_OBJ_NEW_QSTR(module_name));
117 
118  // store the new module into the slot in the global dict holding all modules
119  el->value = MP_OBJ_FROM_PTR(o);
120 
121  // return the new module
122  return MP_OBJ_FROM_PTR(o);
123 }

238 // returns MP_OBJ_NULL if not found
239 mp_obj_t mp_module_get(qstr module_name) {
240  mp_map_t *mp_loaded_modules_map = &MP_STATE_VM(mp_loaded_modules_dict).map;
241  // lookup module
242  mp_map_elem_t *el = mp_map_lookup(mp_loaded_modules_map, MP_OBJ_NEW_QSTR(module_name), MP_MAP_LOOKUP);
243 
244  if (el == NULL) {
245  // module not found, look for builtin module names
246  el = mp_map_lookup((mp_map_t*)&mp_builtin_module_map, MP_OBJ_NEW_QSTR(module_name), MP_MAP_LOOKUP);
247  if (el == NULL) {
248  return MP_OBJ_NULL;
249  }
250 
251  if (MICROPY_MODULE_BUILTIN_INIT) {
252  // look for __init__ and call it if it exists
253  mp_obj_t dest[2];
254  mp_load_method_maybe(el->value, MP_QSTR___init__, dest);
255  if (dest[0] != MP_OBJ_NULL) {
256  mp_call_method_n_kw(0, 0, dest);
257  // register module so __init__ is not called again
258  mp_module_register(module_name, el->value);
259  }
260  }
261  }
262 
263  // module found, return it
264  return el->value;
265 }
