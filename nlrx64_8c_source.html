103 NORETURN void nlr_jump(void *val) {
104  nlr_buf_t **top_ptr = &MP_STATE_THREAD(nlr_top);
105  nlr_buf_t *top = *top_ptr;
106  if (top == NULL) {
107  nlr_jump_fail(val);
108  }
109 
110  top->ret_val = val;
111  *top_ptr = top->prev;
112 
113  __asm volatile (
114  "movq %0, %%rcx \n" // %rcx points to nlr_buf
115  #if NLR_OS_WINDOWS
116  "movq 88(%%rcx), %%rsi \n" // load saved %rsi
117  "movq 80(%%rcx), %%rdi \n" // load saved %rdr
118  #endif
119  "movq 72(%%rcx), %%r15 \n" // load saved %r15
120  "movq 64(%%rcx), %%r14 \n" // load saved %r14
121  "movq 56(%%rcx), %%r13 \n" // load saved %r13
122  "movq 48(%%rcx), %%r12 \n" // load saved %r12
123  "movq 40(%%rcx), %%rbx \n" // load saved %rbx
124  "movq 32(%%rcx), %%rsp \n" // load saved %rsp
125  "movq 24(%%rcx), %%rbp \n" // load saved %rbp
126  "movq 16(%%rcx), %%rax \n" // load saved %rip
127  "movq %%rax, (%%rsp) \n" // store saved %rip to stack
128  "xorq %%rax, %%rax \n" // clear return register
129  "inc %%al \n" // increase to make 1, non-local return
130  "ret \n" // return
131  : // output operands
132  : "r"(top) // input operands
133  : // clobbered registers
134  );
135 
136  for (;;); // needed to silence compiler warning
137 }
