39 void Arithmetic::PrimeFactorization(const Integer * n, Integer * outputFactors, Integer * outputCoefficients, int outputLength) {
40  /* First we look for prime divisors in the table primeFactors (to speed up
41  * the prime factorization for low numbers). When k_numberOfPrimeFactors is
42  * overflow, try every number as divisor. */
43  Integer m = *n;
44  m.setNegative(false);
45  if (m.isEqualTo(Integer(1))) {
46  return;
47  }
48  const Integer primorial32("525896479052627740771371797072411912900610967452630");
49  if (primorial32.isLowerThan(m)) {
50  /* Special case 1: We do not want to break i in prime factor because it
51  * might take too many factors... More than k_maxNumberOfPrimeFactors.
52  * outputCoefficients[0] is set to -1 to indicate a special case. */
53  outputCoefficients[0] = Integer(-1);
54  return;
55  }
56  for (int index = 0; index < outputLength; index++) {
57  outputCoefficients[index] = Integer(0);
58  }
59 
60  int t = 0; // n prime factor index
61  int k = 0; // prime factor index
62  Integer testedPrimeFactor = Integer(primeFactors[k]); // prime factor
63  outputFactors[t] = testedPrimeFactor;
64  IntegerDivision d = {.quotient = 0, .remainder = 0};
65  bool stopCondition;
66  do {
67  stopCondition = Integer::Power(outputFactors[t], Integer(2)).isLowerThan(m);
68  d = Integer::Division(m, testedPrimeFactor);
69  if (d.remainder.isEqualTo(Integer(0))) {
70  outputCoefficients[t] = Integer::Addition(outputCoefficients[t], Integer(1));
71  m = std::move(d.quotient);
72  if (m.isEqualTo(Integer(1))) {
73  return;
74  }
75  continue;
76  }
77  if (!outputCoefficients[t].isEqualTo(Integer(0))) {
78  t++;
79  }
80  k++;
81  testedPrimeFactor = k < k_numberOfPrimeFactors ? Integer(primeFactors[k]) : Integer::Addition(testedPrimeFactor, Integer(1));
82  outputFactors[t] = testedPrimeFactor;
83  } while (stopCondition && testedPrimeFactor.isLowerThan(Integer(k_biggestPrimeFactor)));
84  if (Integer::Power(Integer(k_biggestPrimeFactor), Integer(2)).isLowerThan(m)) {
85  /* Special case 2: We do not want to break i in prime factor because it
86  * take too much time: the prime factor that should be tested is above
87  * k_biggestPrimeFactor.
88  * outputCoefficients[0] is set to -1 to indicate a special case. */
89  outputCoefficients[0] = -1;
90  return;
91  }
92  outputFactors[t] = std::move(m);
93  outputCoefficients[t] = Integer::Addition(outputCoefficients[t], Integer(1));
94 }
95 }
