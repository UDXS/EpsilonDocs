31 // This is universal iterator type which calls "iternext" method stored in
32 // particular object instance. (So, each instance of this time can have its
33 // own iteration behavior.) Having this type saves to define type objects
34 // for various internal iterator objects.
35 
36 // Any instance should have these 2 fields at the beginning
37 typedef struct _mp_obj_polymorph_iter_t {
38  mp_obj_base_t base;
39  mp_fun_1_t iternext;
40 } mp_obj_polymorph_iter_t;

42 STATIC mp_obj_t polymorph_it_iternext(mp_obj_t self_in) {
43  mp_obj_polymorph_iter_t *self = MP_OBJ_TO_PTR(self_in);
44  // Redirect call to object instance's iternext method
45  return self->iternext(self_in);
46 }
