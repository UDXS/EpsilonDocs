46 STATIC mp_obj_t filter_iternext(mp_obj_t self_in) {
47  mp_check_self(MP_OBJ_IS_TYPE(self_in, &mp_type_filter));
48  mp_obj_filter_t *self = MP_OBJ_TO_PTR(self_in);
49  mp_obj_t next;
50  while ((next = mp_iternext(self->iter)) != MP_OBJ_STOP_ITERATION) {
51  mp_obj_t val;
52  if (self->fun != mp_const_none) {
53  val = mp_call_function_n_kw(self->fun, 1, 0, &next);
54  } else {
55  val = next;
56  }
57  if (mp_obj_is_true(val)) {
58  return next;
59  }
60  }
61  return MP_OBJ_STOP_ITERATION;
62 }
