38 STATIC mp_obj_t it_iternext(mp_obj_t self_in) {
39  mp_obj_getitem_iter_t *self = MP_OBJ_TO_PTR(self_in);
40  nlr_buf_t nlr;
41  if (nlr_push(&nlr) == 0) {
42  // try to get next item
43  mp_obj_t value = mp_call_method_n_kw(1, 0, self->args);
44  self->args[2] = MP_OBJ_NEW_SMALL_INT(MP_OBJ_SMALL_INT_VALUE(self->args[2]) + 1);
45  nlr_pop();
46  return value;
47  } else {
48  // an exception was raised
49  mp_obj_type_t *t = (mp_obj_type_t*)((mp_obj_base_t*)nlr.ret_val)->type;
50  if (t == &mp_type_StopIteration || t == &mp_type_IndexError) {
51  // return MP_OBJ_STOP_ITERATION instead of raising
52  return MP_OBJ_STOP_ITERATION;
53  } else {
54  // re-raise exception
55  nlr_jump(nlr.ret_val);
56  }
57  }
58 }
