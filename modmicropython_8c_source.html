67 mp_obj_t mp_micropython_mem_info(size_t n_args, const mp_obj_t *args) {

73 #if MICROPY_STACK_CHECK
74  mp_printf(&mp_plat_print, "stack: " UINT_FMT " out of " UINT_FMT "\n",
75  mp_stack_usage(), (mp_uint_t)MP_STATE_THREAD(stack_limit));

77  mp_printf(&mp_plat_print, "stack: " UINT_FMT "\n", mp_stack_usage());

79 #if MICROPY_ENABLE_GC
80  gc_dump_info();
81  if (n_args == 1) {
82  // arg given means dump gc allocation table
83  gc_dump_alloc_table();
84  }
85 #else
86  (void)n_args;
87 #endif
88  return mp_const_none;
89 }
