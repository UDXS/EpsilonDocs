29  GET_FLOAT_WORD(ix,x);
30  k = (ix&0x7f800000)>>23; /* extract exponent */
31  if (k==0) { /* 0 or subnormal x */
32  if ((ix&0x7fffffff)==0) return x; /* +-0 */
33  x *= two25;
34  GET_FLOAT_WORD(ix,x);
35  k = ((ix&0x7f800000)>>23) - 25;
36  if (n< -50000) return tiny*x; /*underflow*/
37  }
38  if (k==0xff) return x+x; /* NaN or Inf */
39  k = k+n;
40  if (k > 0xfe) return huge*copysignf(huge,x); /* overflow */
41  if (k > 0) /* normal result */
42  {SET_FLOAT_WORD(x,(ix&0x807fffff)|(k<<23)); return x;}
43  if (k <= -25)
