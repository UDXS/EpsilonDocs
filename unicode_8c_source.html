71 unichar utf8_get_char(const byte *s) {
72 #if MICROPY_PY_BUILTINS_STR_UNICODE
73  unichar ord = *s++;
74  if (!UTF8_IS_NONASCII(ord)) return ord;
75  ord &= 0x7F;
76  for (unichar mask = 0x40; ord & mask; mask >>= 1) {
77  ord &= ~mask;
78  }
79  while (UTF8_IS_CONT(*s)) {
80  ord = (ord << 6) | (*s++ & 0x3F);
81  }
82  return ord;
83 #else
84  return *s;
85 #endif
86 }

186 bool utf8_check(const byte *p, size_t len) {
187  uint8_t need = 0;
188  const byte *end = p + len;
189  for (; p < end; p++) {
190  byte c = *p;
191  if (need) {
192  if (c >= 0x80) {
193  need--;
194  } else {
195  // mismatch
196  return 0;
197  }
198  } else {
199  if (c >= 0xc0) {
200  if (c >= 0xf8) {
201  // mismatch
202  return 0;
203  }
204  need = (0xe5 >> ((c >> 3) & 0x6)) & 3;
205  } else if (c >= 0x80) {
206  // mismatch
207  return 0;
208  }
209  }
210  }
211  return need == 0; // no pending fragments allowed
212 }
