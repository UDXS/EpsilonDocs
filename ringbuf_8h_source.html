48 static inline int ringbuf_get(ringbuf_t *r) {
49  if (r->iget == r->iput) {
50  return -1;
51  }
52  uint8_t v = r->buf[r->iget++];
53  if (r->iget >= r->size) {
54  r->iget = 0;
55  }
56  return v;
57 }
58 
59 static inline int ringbuf_put(ringbuf_t *r, uint8_t v) {
60  uint32_t iput_new = r->iput + 1;
61  if (iput_new >= r->size) {
62  iput_new = 0;
63  }
64  if (iput_new == r->iget) {
65  return -1;
66  }
67  r->buf[r->iput] = v;
68  r->iput = iput_new;
69  return 0;
70 }
