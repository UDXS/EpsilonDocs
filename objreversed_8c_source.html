58 STATIC mp_obj_t reversed_iternext(mp_obj_t self_in) {
59  mp_check_self(MP_OBJ_IS_TYPE(self_in, &mp_type_reversed));
60  mp_obj_reversed_t *self = MP_OBJ_TO_PTR(self_in);
61 
62  // "raise" stop iteration if we are at the end (the start) of the sequence
63  if (self->cur_index == 0) {
64  return MP_OBJ_STOP_ITERATION;
65  }
66 
67  // pre-decrement and index sequence
68  self->cur_index -= 1;
69  return mp_obj_subscr(self->seq, MP_OBJ_NEW_SMALL_INT(self->cur_index), MP_OBJ_SENTINEL);
70 }
